SQLite from Python

import sqlite3
conn = sqlite3.connect("app.db")
cursor = conn.cursor()

cursor.execute("""CREATE TABLE IF NOT EXISTS users (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT
)""")

# Always use ? placeholders!
cursor.execute("INSERT INTO users (name) VALUES (?)", ("Armaghan",))
conn.commit()
conn.close()

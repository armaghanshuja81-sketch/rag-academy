Document Chunking

Splitting documents into smaller pieces for better retrieval.

def chunk_recursive(text, size=500):
    paragraphs = text.split("\n\n")
    chunks, current = [], ""
    for p in paragraphs:
        if len(current) + len(p) < size:
            current += p + "\n"
        else:
            chunks.append(current.strip())
            current = p + "\n"
    if current: chunks.append(current.strip())
    return chunks

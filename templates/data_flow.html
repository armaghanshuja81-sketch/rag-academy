🔬 The Flask Code That Makes This Work

Python (Flask Route) 
@app.route('/data-flow', methods=['POST'])
def data_flow_submit():
    """The FULL data flow demonstrated in one route."""

    # STEP 1: Browser sends data as HTTP POST request
    # Flask receives it in request.form (a dictionary)
    name = request.form['name']       # ← From name="name" in HTML
    topic = request.form['topic']     # ← From name="topic" in HTML
    message = request.form['message'] # ← From name="message" in HTML

    # STEP 2: Flask processes the data (business logic)
    # We create a formatted response
    response_text = f"{name} asked about {topic}: {message}"

    # STEP 3: Flask SAVES data to the DATABASE
    conn = sqlite3.connect('rag_academy.db')
    conn.execute(
        "INSERT INTO user_queries (name, topic, message) VALUES (?, ?, ?)",
        (name, topic, message)
    )
    conn.commit()
    conn.close()

    # STEP 4: Flask sends HTML back as HTTP Response
    return render_template('data_flow.html',
        last_submission=response_text,
        data_flow_trace=trace_data,
        flow_steps=flow_steps
    )
    # Browser renders the HTML → You see the result!
